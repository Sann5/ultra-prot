def impute_downshifted_normal_global(
        adata,
        scale=0.3,
        shift=1.8,
):
    np.random.seed(42)
    mean = np.nanmean(adata.X)
    std = np.nanstd(adata.X)
    draws = np.random.normal(loc=mean - shift * std, scale=scale * std, size=np.sum(np.isnan(adata.X)))
    adata.X[np.isnan(adata.X)] = draws
